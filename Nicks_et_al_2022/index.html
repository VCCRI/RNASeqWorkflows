# Differential expression ysing EdgeR

geneCounts <- as.matrix(count_matrix_P2)


library("edgeR")
# filter out low expressed transcripts
numberSamples=4
countThreshold=1
padj_threshold=0.01

isexpr <- rowSums(cpm(geneCounts) >= countThreshold) >= numberSamples
geneCounts.expressed <- geneCounts[isexpr, ]
dim(geneCounts.expressed)
head(geneCounts.expressed)

#            '1'    '2'    '3'    '4'    '5'   '6'    '19'  '20'
conds <- c('P2M', 'P2F', 'P2M', 'P2F', 'P2M','P2F', 'P2M','P2F')
condition <- factor(conds)

y <- DGEList(counts = matrix(geneCounts.expressed, ncol = 8, 
        dimnames = list(row.names(geneCounts.expressed))), group = condition)

y$samples$lib.size <- colSums(y$counts)
y <- calcNormFactors(y)
y <- estimateCommonDisp(y)
y <- estimateTagwiseDisp(y)
et <- exactTest(y, pair = c('P2F', 'P2M'))
edge <- as.data.frame(topTags(et, n = 1e+06))
edge2fold <- edge[edge$logFC >= FC_threshold | edge$logFC <= (-1 * FC_threshold), ]
edge2foldpadj <- edge2fold[edge2fold$FDR <= padj_threshold, ]
